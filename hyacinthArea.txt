function classification(image){
  
  var label = 'landcover'
  var bands = ['B1', 'B2', 'B3', 'B4', 'B5', 'B6','B7']

  var training = image.select(bands).sampleRegions({
    collection: avg,
    properties: [label],
    scale: 30
  });
  
  var trained = ee.Classifier.smileRandomForest(10).train(training, label, bands);
  var classified = image.select(bands).classify(trained);
  
  return classified
  
}

var classified_l7 = dataset.map(classification)

var hyacinth=classified_l7.map(function(image){
  var  hyacinthmask=image.select('classification').eq(1);
  image.updateMask(hyacinthmask);

  var hyacinthArea = hyacinthmask.multiply(area).rename('HyacinthArea');

  image = image.addBands(hyacinthArea);

  var stats = hyacinthArea.reduceRegion({
    reducer: ee.Reducer.sum(), 
    geometry: lV, 
    scale: 30,

  return image.set(stats);
})
